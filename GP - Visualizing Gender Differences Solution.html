fig = plt.figure(figsize=(15, 18))

for outer_counter, outer_list_name in enumerate(cats): #Elements of outer list are the three inner lists of degree categories
    for inner_counter, inner_list_value in enumerate(cats[outer_counter]): #Elements of each of the three inner list are
                                                                     #the degree descriptions
        ax = fig.add_subplot(6, 3, outer_counter + (inner_counter * 3) + 1)
        ax.plot(women_degrees['Year'], women_degrees[cats[outer_counter][inner_counter]], 
                c = cb_dark_blue, label = 'Women', linewidth = 3)
        ax.plot(women_degrees['Year'], 100-women_degrees[cats[outer_counter][inner_counter]],
                c = cb_orange, label='Men', linewidth = 3)
        ax.spines["right"].set_visible(False)    
        ax.spines["left"].set_visible(False)
        ax.spines["top"].set_visible(False)    
        ax.spines["bottom"].set_visible(False)
        ax.set_xlim(1968, 2011)
        ax.set_ylim(0,100)
        ax.set_yticks([0,100])
        ax.axhline(50, c=(171/255, 171/255, 171/255), alpha=0.3)
        ax.set_title(cats[outer_counter][inner_counter])
        ax.tick_params(bottom = False, top = False, left = False, right = False, labelbottom=False)
        
        if (outer_counter == 0) and (inner_counter == 0):
            ax.text(2003, 87, 'Women')
            ax.text(2005, 8, 'Men')
        elif (outer_counter == 0) and (inner_counter == 5):
            ax.text(2003, 6, 'Women')
            ax.text(2005, 90, 'Men')
            ax.tick_params(labelbottom=True)
